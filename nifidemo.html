Java — Custom POS Producer Processor PRODUCER 1
@Tags({"grocery", "pos", "producer"})
@CapabilityDescription("Reads POS scan events and emits FlowFiles")
@SupportsBatching
public class POSScannerProcessor extends AbstractProcessor {

    static final PropertyDescriptor STORE_ID = new PropertyDescriptor.Builder()
        .name("store-id")
        .displayName("Store ID")
        .description("Unique identifier for the grocery store location")
        .required(true)
        .defaultValue("STORE-042")
        .addValidator(StandardValidators.NON_EMPTY_VALIDATOR)
        .build();

    static final Relationship SUCCESS = new Relationship.Builder()
        .name("success")
        .description("Scan event successfully parsed")
        .build();

    static final Relationship FAILURE = new Relationship.Builder()
        .name("failure")
        .description("Malformed scan event")
        .build();

    @Override
    public void onTrigger(ProcessContext ctx, ProcessSession session) {
        FlowFile flowFile = session.create();

        String storeId  = ctx.getProperty(STORE_ID).getValue();
        String category = detectCategory(flowFile); // produce/dairy/frozen/deli

        // Stamp attributes for downstream routing
        flowFile = session.putAttribute(flowFile, "store.id",      storeId);
        flowFile = session.putAttribute(flowFile, "item.category", category);
        flowFile = session.putAttribute(flowFile, "source.type",   "pos_scanner");
        flowFile = session.putAttribute(flowFile, "event.time",
            Instant.now().toString());

        // Write JSON payload
        flowFile = session.write(flowFile, out -> {
            String json = buildScanJson(storeId, category);
            out.write(json.getBytes(StandardCharsets.UTF_8));
        });

        session.transfer(flowFile, SUCCESS);
        session.commitAsync();
    }

    private String buildScanJson(String storeId, String category) {
        return String.format("""
            {"transaction_id":"%s","store_id":"%s",
             "category":"%s","timestamp":"%s"}
            """,
            UUID.randomUUID(), storeId, category, Instant.now());
    }
}

🔃 RouteOnAttribute — Fan-out to 3 Consumers PROCESSOR
Routes each FlowFile to one or more consumers based on item.category and source.type. All events go to Analytics; only POS sales go to Billing; all stock events go to Inventory. 
// Processor: RouteOnAttribute
// NiFi Expression Language routing rules:

"to_inventory"  : "${source.type:equals('pos_scanner'):or(${source.type:equals('supplier_feed')})}"
"to_billing"    : "${source.type:equals('pos_scanner')}"
"to_analytics"  : "${literal(true)}"   // ALL events → analytics

// Routing Strategy: Route to Property Name
// Unmatched → "unmatched" relationship (send to dead-letter queue)

💳 BillingSystem — PutHTTP (Java Custom) CONSUMER 2
POSTs completed sale transactions to the external billing/POS settlement API over HTTPS. Only receives FlowFiles where source.type = pos_scanner. 
// Processor: PutHTTP (or custom InvokeHTTP)
HTTP Method:        POST
Remote URL:         "https://billing.internal/api/v2/transactions"
Content-Type:       "application/json"
Request Header:     "X-Store-ID: ${store.id}"
SSL Context Service: StandardSSLContextService
Connection Timeout: 5 sec
Read Timeout:       10 sec

// Custom Java processor for retry + idempotency key:
@Tags({"billing", "http", "grocery"})
public class BillingHttpConsumer extends AbstractProcessor {

    @Override
    public void onTrigger(ProcessContext ctx, ProcessSession session) {
        FlowFile ff = session.get();
        if (ff == null) return;

        String txnId   = ff.getAttribute("event.id");
        String storeId = ff.getAttribute("store.id");

        try {
            HttpResponse resp = postToBilling(ff, txnId, storeId, session);

            if (resp.statusCode() == 200 || resp.statusCode() == 201) {
                ff = session.putAttribute(ff, "billing.status", "settled");
                session.transfer(ff, SUCCESS);
            } else if (resp.statusCode() == 409) {
                // Idempotency: already processed
                session.transfer(ff, SUCCESS);
            } else {
                session.penalize(ff);
                session.transfer(ff, RETRY);
            }
        } catch (IOException e) {
            session.penalize(ff);
            session.transfer(ff, RETRY);
        }
    }
}
