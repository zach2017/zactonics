Ansible Basics

Ansible is an agentless push-model automation engine. A control node SSHes into managed nodes and executes Python modules. No daemon runs on targets — each connection is ephemeral and stateless.

🔀 Part 02

When & Conditionals

The when clause uses Jinja2 expressions evaluated per host. When false, the task is skipped — not failed. Combine multiple conditions with a YAML list (implicit AND) or Jinja2 or.

yaml
# ── 1. OS-family conditionals ─────────────────────────────────
- name: Install Java (Debian/Ubuntu)
  apt:
    name: openjdk-21-jdk-headless
    state: present
  when: ansible_os_family == "Debian"

- name: Install Java (RHEL/Rocky)
  dnf:
    name: java-21-openjdk-headless
    state: present
  when: ansible_os_family == "RedHat"

# ── 2. Multiple conditions (AND) ──────────────────────────────
- name: Enable ZooKeeper TLS
  template:
    src: zoo-tls.cfg.j2
    dest: "{{ zk_conf_dir }}/zoo.cfg"
  when:
    - zookeeper_tls_enabled | bool
    - zookeeper_cert_file is defined
    - zookeeper_cert_file | length > 0

# ── 3. register + when — react to command output ──────────────
- name: Check if myid file exists
  stat:
    path: "{{ zookeeper_data_dir }}/myid"
  register: myid_stat

- name: Write myid (first install only)
  copy:
    content: "{{ zookeeper_myid }}"
    dest: "{{ zookeeper_data_dir }}/myid"
  when: not myid_stat.stat.exists

# ── 4. Compare version strings ────────────────────────────────
- name: Read installed ZK version
  shell: >
    {{ zookeeper_install_dir }}/bin/zkServer.sh version 2>&1
    | grep -oP 'version \K[\d.]+'
  register: zk_current_ver
  changed_when: false
  failed_when: false

- name: Run upgrade only when version differs
  include_tasks: upgrade.yml
  when:
    - zk_current_ver.rc == 0
    - zk_current_ver.stdout != zookeeper_version

# ── 5. Conditional block ──────────────────────────────────────
- name: Monitoring block (conditional)
  block:
    - name: Deploy Prometheus exporter
      template:
        src: zk-exporter.service.j2
        dest: /etc/systemd/system/zk-exporter.service
    - name: Start exporter
      systemd:
        name: zk-exporter
        enabled: yes
        state: started
  when: zookeeper_metrics_enabled | default(false) | bool
✅ Use changed_when: false on read-only commands so they never show as "changed" in your reports.

# ── Health check → conditional restart ───────────────────────
- name: Check Elasticsearch cluster health
  uri:
    url: "https://{{ ansible_host }}:9200/_cluster/health"
    user: "{{ es_user }}"
    password: "{{ es_password }}"
    validate_certs: "{{ es_validate_certs }}"
    return_content: yes
  register: es_health
  changed_when: false
  failed_when: false

- name: Fail if cluster is RED
  fail:
    msg: "Cluster health is RED — aborting deployment"
  when: >
    es_health.status == 200 and
    es_health.json.status | default('unknown') == 'red'

- name: Warn if cluster is YELLOW
  debug:
    msg: "Warning: {{ es_health.json.unassigned_shards }} unassigned shards"
  when: >
    es_health.status == 200 and
    es_health.json.status | default('') == 'yellow'

- name: Validate required variables
  assert:
    that:
      - etl_batch_size | int > 0
      - etl_batch_size | int <= 10000
      - zookeeper_client_port | int == 2181

    fail_msg: >
      Validation failed:
        etl_batch_size={{ etl_batch_size }} (must be 1-10000)
        es_index_pattern={{ es_index_pattern }} (lowercase only)

- name: List active ETL services

- name: Build ZooKeeper connection string
  set_fact:
    zk_connect: >-
      {{ groups['zookeeper']

📋 Part 03

Copy, Logs & Debug

copy pushes files from control → remote. fetch pulls from remote → control. Use archive + fetch to collect compressed log bundles.

yaml
# ── copy: push files to managed nodes ────────────────────────
- name: Copy TLS certificate to all nodes
  copy:
    src: files/certs/es-ca.crt
    dest: /etc/ssl/certs/es-ca.crt
    owner: root
    group: root
    mode: "0644"
    backup: yes                   # keeps timestamped backup
  notify: Restart ETL workers

- name: Write inline MOTD banner
  copy:
    content: |
      ──────────────────────────────────────
      Production Node — Managed via Ansible
      Env: {{ env }}  Role: {{ node_role }}
      ──────────────────────────────────────
    dest: /etc/motd

- name: Sync ETL application directory
  synchronize:
    src: "{{ playbook_dir }}/files/etl_app/"
    dest: "{{ etl_install_dir }}/src/"
    delete: yes
    rsync_opts:
      - "--exclude=*.pyc"
      - "--exclude=__pycache__"
      - "--exclude=.env"
  notify: Restart ETL workers

# ── fetch: pull logs back to control node ────────────────────
- name: Fetch ZooKeeper log from each node
  fetch:
    src: "{{ zookeeper_log_dir }}/zookeeper.log"
    dest: "./fetched_logs/{{ inventory_hostname }}/zk.log"
    flat: yes

- name: Fetch ETL worker logs
  fetch:
    src: "{{ etl_log_dir }}/worker-{{ item }}.log"
    dest: "./fetched_logs/{{ inventory_hostname }}/worker-{{ item }}.log"
    flat: yes
    fail_on_missing: no
  loop: [1, 2, 3, 4]

# ── archive + fetch ───────────────────────────────────────────
- name: Compress logs before pulling
  archive:
    path: "{{ zookeeper_log_dir }}/*.log"
    dest: "/tmp/zk-logs-{{ ansible_date_time.date }}.tar.gz"
    format: gz
  register: zk_archive

- name: Pull compressed archive
  fetch:
    src: "{{ zk_archive.dest }}"
    dest: "./log_archives/"
    flat: yes

- name: Clean up remote archive
  file:
    path: "{{ zk_archive.dest }}"
    state: absent

yaml
- name: Safe ETL deployment with automatic rollback
  block:
    # ── try ──────────────────────────────────────────────────
    - name: Stop workers
      systemd:
        name: "etl-worker@{{ item }}"
        state: stopped
      loop: "{{ range(1, etl_workers + 1) | list }}"

    - name: Deploy new version
      synchronize:
        src: "files/etl_app_{{ etl_app_version }}/"
        dest: "{{ etl_install_dir }}/src/"
        delete: yes

    - name: Run smoke test
      command: "{{ etl_venv_dir }}/bin/python src/smoke_test.py"
      args:
        chdir: "{{ etl_install_dir }}"
      register: smoke
      failed_when: smoke.rc != 0

    - name: Start workers
      systemd:
        name: "etl-worker@{{ item }}"
        state: started
      loop: "{{ range(1, etl_workers + 1) | list }}"

  rescue:
    # ── catch ────────────────────────────────────────────────
    - name: Log failure
      lineinfile:
        path: "{{ etl_log_dir }}/deployments.log"
        line: "[{{ ansible_date_time.iso8601 }}] FAILED version={{ etl_app_version }}"
        create: yes

    - name: Roll back to previous version
      synchronize:
        src: "{{ etl_install_dir }}/src_backup/"
        dest: "{{ etl_install_dir }}/src/"
        delete: yes

    - name: Restart workers on old version
      systemd:
        name: "etl-worker@{{ item }}"
        state: restarted
      loop: "{{ range(1, etl_workers + 1) | list }}"

    - name: Alert on-call via webhook
      uri:
        url: "{{ alert_webhook_url }}"
        method: POST
        body_format: json
        body:
          text: "Deployment FAILED on {{ inventory_hostname }} — rolled back"
          severity: high

  always:
    # ── finally ──────────────────────────────────────────────
    - name: Collect post-deploy logs regardless of outcome
      fetch:
        src: "{{ etl_log_dir }}/worker-1.log"
        dest: "./fetched_logs/{{ inventory_hostname }}-post-deploy.log"
        flat: yes
        fail_on_missing: no

# roles/zookeeper/tasks/main.yml
---
- name: Create system group
  group:
    name: "{{ zookeeper_group }}"
    state: present

- name: Create system user
  user:
    name: "{{ zookeeper_user }}"
    group: "{{ zookeeper_group }}"
    system: yes
    shell: /sbin/nologin

- name: Create directories
  file:
    path: "{{ item }}"
    state: directory
    owner: "{{ zookeeper_user }}"
    group: "{{ zookeeper_group }}"
    mode: "0755"
  loop:
    - "{{ zookeeper_install_dir }}"
    - "{{ zookeeper_data_dir }}"
    - "{{ zookeeper_log_dir }}"

- name: Download tarball
  get_url:
    url: "{{ zookeeper_mirror }}/zookeeper-{{ zookeeper_version }}/apache-zookeeper-{{ zookeeper_version }}-bin.tar.gz"
    dest: "/tmp/zookeeper-{{ zookeeper_version }}.tar.gz"

- name: Extract ZooKeeper
  unarchive:
    src: "/tmp/zookeeper-{{ zookeeper_version }}.tar.gz"
    dest: "{{ zookeeper_install_dir }}"
    remote_src: yes
    extra_opts: ["--strip-components=1"]
    owner: "{{ zookeeper_user }}"
    creates: "{{ zookeeper_install_dir }}/bin/zkServer.sh"

- name: Write myid
  copy:
    content: "{{ zookeeper_myid }}"
    dest: "{{ zookeeper_data_dir }}/myid"
  when: not (zookeeper_data_dir + '/myid') is exists

- name: Deploy zoo.cfg
  template:
    src: zoo.cfg.j2
    dest: "{{ zookeeper_install_dir }}/conf/zoo.cfg"
    owner: "{{ zookeeper_user }}"
  notify: Restart ZooKeeper

- name: Deploy systemd unit
  template:
    src: zookeeper.service.j2
    dest: /etc/systemd/system/zookeeper.service
  notify:
    - Reload systemd
    - Restart ZooKeeper

- name: Enable and start ZooKeeper
  systemd:
    name: zookeeper
    enabled: yes
    state: started

public class ElasticsearchSource implements AutoCloseable {

    private static final Logger log =
        LoggerFactory.getLogger(ElasticsearchSource.class);

    private final ElasticsearchClient esClient;
    private final RestClient restClient;
    private final EtlConfig config;

    public ElasticsearchSource(EtlConfig config) {
        this.config = config;
        var creds = new BasicCredentialsProvider();
        creds.setCredentials(
            org.apache.http.auth.AuthScope.ANY,
            new UsernamePasswordCredentials(
                config.esUser(), config.esPassword()));

        this.restClient = RestClient
            .builder(config.esHosts().stream()
                .map(h -> new HttpHost(h.host(), h.port(), "https"))
                .toArray(HttpHost[]::new))
            .setHttpClientConfigCallback(b -> b
                .setDefaultCredentialsProvider(creds)
                .setSSLContext(config.sslContext()))
            .setRequestConfigCallback(r -> r
                .setConnectTimeout(5_000)
                .setSocketTimeout(60_000))
            .build();

        this.esClient = new ElasticsearchClient(
            new RestClientTransport(restClient, new JacksonJsonpMapper()));
    }

    /**
     * Paginate all documents using PIT + search_after.
     * More efficient than scroll — no server-side cursor state.
     */
    public List<Map> extractPartition(
            int partitionId, Instant from, Instant to) throws IOException {

        var results = new ArrayList<Map>();

        // Open Point In Time for consistent reads
        var pitId = esClient.openPointInTime(p -> p
            .index(config.esIndex())
            .keepAlive(t -> t.time("2m"))).id();

        try {
            List<String> searchAfter = null;
            int batch = 0;

            while (true) {
                final var sa = searchAfter;
                var req = SearchRequest.of(r -> {
                    r.pit(p -> p.id(pitId).keepAlive(t -> t.time("2m")))
                     .size(config.scrollSize())
                     .sort(s -> s.field(f -> f
                         .field("@timestamp").order(SortOrder.Asc)))
                     .sort(s -> s.field(f -> f
                         .field("_shard_doc").order(SortOrder.Asc)))
                     .query(q -> q.bool(b -> b
                         .filter(f -> f.range(rng -> rng
                             .field("@timestamp")
                             .from(from.toString())
                             .to(to.toString())))
                         .filter(f -> f.term(t -> t
                             .field("partition")
                             .value(partitionId)))));
                    if (sa != null) r.searchAfter(sa);
                    return r;
                });

                var resp = esClient.search(req, Map.class);
                var hits = resp.hits().hits();
                if (hits.isEmpty()) break;

                hits.forEach(h -> results.add(h.source()));

                var last = hits.get(hits.size() - 1);
                searchAfter = last.sort().stream()
                    .map(Object::toString).toList();

                log.info("Partition {} batch {} — {} docs (total {})",
                    partitionId, ++batch, hits.size(), results.size());
            }
        } finally {
            esClient.closePointInTime(c -> c.id(pitId));
        }
        return results;
    }

    /** Terms aggregation: count events by type */
    public Map<String, Long> aggregateByType(
            Instant from, Instant to) throws IOException {
        var resp = esClient.search(r -> r
            .index(config.esIndex()).size(0)
            .query(q -> q.range(rng -> rng
                .field("@timestamp")
                .from(from.toString()).to(to.toString())))
            .aggregations("by_type", a -> a
                .terms(t -> t.field("event_type").size(50))),
            Void.class);

        var result = new LinkedHashMap<String, Long>();
        resp.aggregations().get("by_type").sterms()
            .buckets().array()
            .forEach(b -> result.put(b.key().stringValue(), b.docCount()));
        return result;
    }

    @Override
    public void close() throws Exception { restClient.close(); }
}

package com.example.etl;

import org.apache.curator.framework.*;
import org.apache.curator.framework.recipes.leader.*;
import org.apache.curator.framework.recipes.locks.InterProcessMutex;
import org.apache.curator.retry.ExponentialBackoffRetry;
import org.apache.zookeeper.CreateMode;
import org.apache.zookeeper.KeeperException.NodeExistsException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicBoolean;

public class ZooKeeperCoordinator implements AutoCloseable {

    private static final Logger log =
        LoggerFactory.getLogger(ZooKeeperCoordinator.class);

    private final CuratorFramework curator;
    private final String workerId;
    private LeaderLatch leaderLatch;
    private final AtomicBoolean isLeader = new AtomicBoolean(false);

    public ZooKeeperCoordinator(String zkConnect, String workerId,
                                 String namespace) {
        this.workerId = workerId;
        this.curator = CuratorFrameworkFactory.builder()
            .connectString(zkConnect)
            .sessionTimeoutMs(10_000)
            .connectionTimeoutMs(5_000)
            .retryPolicy(new ExponentialBackoffRetry(1_000, 5))
            .namespace(namespace.replaceFirst("^/", ""))
            .build();
        curator.start();
        log.info("Connected to ZooKeeper: {}", zkConnect);
    }

    // ── Leader election ────────────────────────────────────────
    public void startLeaderElection(String path) throws Exception {
        leaderLatch = new LeaderLatch(curator, path, workerId);
        leaderLatch.addListener(new LeaderLatchListener() {
            @Override public void isLeader() {
                isLeader.set(true);
                log.info("Worker {} became LEADER", workerId);
                onBecomeLeader();
            }
            @Override public void notLeader() {
                isLeader.set(false);
                log.info("Worker {} is FOLLOWER", workerId);
            }
        });
        leaderLatch.start();
    }

    protected void onBecomeLeader() { /* override: rebalance partitions */ }
    public boolean isLeader() { return isLeader.get(); }

    // ── Partition claim (ephemeral znode) ──────────────────────
    public int claimPartition(int maxPartitions) throws Exception {
        for (int p = 0; p < maxPartitions; p++) {
            String path = "/partitions/" + String.format("%04d", p);
            try {
                curator.create()
                    .creatingParentsIfNeeded()
                    .withMode(CreateMode.EPHEMERAL)
                    .forPath(path, workerId.getBytes());
                log.info("Claimed partition {}", p);
                return p;
            } catch (NodeExistsException ignored) { /* try next */ }
        }
        return -1; // none available
    }

    // ── Distributed mutex ──────────────────────────────────────
    public void withLock(String name, Runnable action) throws Exception {
        var mutex = new InterProcessMutex(curator, "/locks/" + name);
        if (!mutex.acquire(30, TimeUnit.SECONDS))
            throw new IllegalStateException("Lock '" + name + "' timeout");
        try {
            log.info("Lock '{}' acquired by {}", name, workerId);
            action.run();
        } finally {
            mutex.release();
        }
    }

    // ── Checkpointing ──────────────────────────────────────────
    public void saveCheckpoint(int partition, String value) throws Exception {
        String path = "/checkpoints/" + partition;
        byte[] data = value.getBytes();
        if (curator.checkExists().forPath(path) == null)
            curator.create().creatingParentsIfNeeded().forPath(path, data);
        else
            curator.setData().forPath(path, data);
    }

    public String loadCheckpoint(int partition) throws Exception {
        String path = "/checkpoints/" + partition;
        if (curator.checkExists().forPath(path) == null) return null;
        return new String(curator.getData().forPath(path));
    }

    @Override public void close() throws Exception {
        if (leaderLatch != null) leaderLatch.close();
        curator.close();
    }
}

    /** Inspect table columns via JDBC DatabaseMetaData */
    public List<ColumnInfo> getColumnMetadata(
            String schema, String table) throws SQLException {
        var cols = new ArrayList<ColumnInfo>();
        try (var conn = ds.getConnection()) {
            var meta = conn.getMetaData();
            try (var rs = meta.getColumns(null, schema, table, "%")) {
                while (rs.next()) {
                    cols.add(new ColumnInfo(
                        rs.getString("COLUMN_NAME"),
                        rs.getString("TYPE_NAME"),
                        rs.getInt("COLUMN_SIZE"),
                        rs.getInt("NULLABLE") == DatabaseMetaData.columnNullable,
                        rs.getString("COLUMN_DEF"),
                        rs.getInt("ORDINAL_POSITION")
                    ));
                }
            }
        }
        return cols;
    }

    /** List all tables in a schema */
    public List<String> listTables(String schema) throws SQLException {
        var tables = new ArrayList<String>();
        try (var conn = ds.getConnection()) {
            try (var rs = conn.getMetaData()
                    .getTables(null, schema, "%", new String[]{"TABLE"})) {
                while (rs.next())
                    tables.add(rs.getString("TABLE_NAME"));
            }
        }
        return tables;
    }

    /** Bulk-insert ETL events with ON CONFLICT DO NOTHING */
    public int bulkInsert(List<EtlEvent> events) throws SQLException {
        final String sql =
            "INSERT INTO events (ts,event_type,user_id,partition,payload) " +
            "VALUES (?,?,?,?,?::jsonb) ON CONFLICT DO NOTHING";
        try (var conn = ds.getConnection();
             var ps   = conn.prepareStatement(sql)) {
            for (var e : events) {
                ps.setObject(1, e.timestamp(), Types.TIMESTAMP_WITH_TIMEZONE);
                ps.setString(2, e.eventType());
                ps.setString(3, e.userId());
                ps.setInt(4, e.partition());
                ps.setString(5, e.payloadJson());
                ps.addBatch();
            }
            int[] counts = ps.executeBatch();
            conn.commit();
            return Arrays.stream(counts)
                .filter(c -> c != Statement.EXECUTE_FAILED).sum();
        }
    }

    public Optional<String> getWatermark(String src, int partition)

                "WHERE source_name=? AND partition_id=?")) {
            ps.setString(1, src); ps.setInt(2, partition);
            try (var rs = ps.executeQuery()) {
                return rs.next()
                    ? Optional.of(rs.getString(1)) : Optional.empty();
            }
        }
    }

"""
worker.py — production ETL worker
  ZooKeeper partition ownership · ES scroll · PG destination
  Prometheus metrics · graceful shutdown · checkpoint resume
"""
import argparse, json, logging, signal, time, yaml
from datetime import datetime, timezone, timedelta
from kazoo.client import KazooClient
from kazoo.exceptions import NodeExistsError
from elasticsearch import Elasticsearch
import psycopg2, psycopg2.extras
from prometheus_client import start_http_server, Counter, Histogram, Gauge

ROWS_OK  = Counter('etl_rows_total',    'Rows written', ['partition'])
DURATION = Histogram('etl_batch_sec',   'Batch latency', ['partition'])
LAG_SEC  = Gauge('etl_lag_seconds',     'Seconds behind now', ['partition'])

class ETLWorker:
    def __init__(self, cfg, worker_id):
        self.cfg, self.worker_id = cfg, worker_id
        self.running = True
        signal.signal(signal.SIGTERM, lambda *_: setattr(self,'running',False))
        signal.signal(signal.SIGINT,  lambda *_: setattr(self,'running',False))

    # ── ZooKeeper ──────────────────────────────────────────────
    def connect_zk(self):
        self.zk = KazooClient(hosts=self.cfg['zookeeper']['hosts'])
        self.zk.start()

    def claim_partition(self):
        root = self.cfg['zookeeper']['root_path']
        for p in range(self.cfg['zookeeper'].get('max_partitions', 100)):
            try:
                self.zk.create(
                    f"{root}/partition-{p:04d}",
                    value=str(self.worker_id).encode(),
                    ephemeral=True, makepath=True)
                logging.info("Claimed partition %d", p)
                return p
            except NodeExistsError:
                continue
        raise RuntimeError("No partitions available")

    def save_checkpoint(self, p, ts):
        path = f"{self.cfg['zookeeper']['root_path']}/ckpt-{p:04d}"
        data = ts.encode()
        (self.zk.set if self.zk.exists(path) else
         lambda *a, **k: self.zk.create(path, data, makepath=True))(path, data)

    def load_checkpoint(self, p):
        path = f"{self.cfg['zookeeper']['root_path']}/ckpt-{p:04d}"
        if not self.zk.exists(path): return None
        data, _ = self.zk.get(path)
        return data.decode() if data else None

    # ── Elasticsearch ──────────────────────────────────────────
    def connect_es(self):
        ec = self.cfg['elasticsearch']
        self.es = Elasticsearch(ec['hosts'],
            http_auth=(ec['username'], ec['password']),
            use_ssl=ec.get('use_ssl', True),
            ca_certs=ec.get('ca_cert'),
            retry_on_timeout=True, max_retries=3)

    def scroll(self, partition, from_ts):
        ec = self.cfg['elasticsearch']
        query = {"query": {"bool": {"filter": [
            {"range":  {"@timestamp": {"gte": from_ts, "lt": "now"}}},
            {"term":   {"partition":  partition}},
        ]}}, "sort": [{"@timestamp": "asc"}, {"_id": "asc"}]}
        r = self.es.search(index=ec['index'], body=query,
                           scroll=ec.get('scroll_timeout','2m'),
                           size=ec.get('scroll_size', 500))
        sid, hits = r['_scroll_id'], r['hits']['hits']
        while hits:
            yield hits
            r    = self.es.scroll(scroll_id=sid,
                                  scroll=ec.get('scroll_timeout','2m'))
            sid  = r['_scroll_id']
            hits = r['hits']['hits']
        self.es.clear_scroll(scroll_id=sid)

    # ── PostgreSQL ─────────────────────────────────────────────
    def connect_pg(self):
        dc = self.cfg['destination']
        self.pg = psycopg2.connect(
            host=dc['host'], port=dc.get('port',5432),
            dbname=dc['database'], user=dc['username'],
            password=dc['password'],
            sslmode=dc.get('sslmode','verify-full'),
            application_name=f"etl-worker-{self.worker_id}")

    def bulk_insert(self, rows):
        if not rows: return 0
        sql = """
            INSERT INTO events (ts,event_type,user_id,partition,payload)
            VALUES (%(ts)s,%(event_type)s,%(user_id)s,
                    %(partition)s,%(payload)s::jsonb)
            ON CONFLICT DO NOTHING
        """
        with self.pg.cursor() as cur:
            psycopg2.extras.execute_batch(cur, sql, rows, page_size=500)
        self.pg.commit()
        return len(rows)

    # ── Main loop ──────────────────────────────────────────────
    def run(self):
        self.connect_zk()
        self.connect_es()
        self.connect_pg()
        start_http_server(
            self.cfg.get('metrics_port', 9000) + self.worker_id)

        p = self.claim_partition()
        ckpt = (self.load_checkpoint(p) or
                (datetime.now(tz=timezone.utc)
                 - timedelta(hours=24)).isoformat())
        logging.info("Starting from %s", ckpt)

        while self.running:
            rows, latest_ts, t0 = [], ckpt, time.time()
            for hits in self.scroll(p, ckpt):
                for h in hits:
                    s = h['_source']
                    rows.append({'ts': s['@timestamp'],
                                 'event_type': s.get('event_type','unknown'),
                                 'user_id': s.get('user_id'),
                                 'partition': p,
                                 'payload': json.dumps(s.get('payload',{}))})
                    if s['@timestamp'] > latest_ts:
                        latest_ts = s['@timestamp']
                if len(rows) >= 500:
                    ROWS_OK.labels(partition=p).inc(self.bulk_insert(rows))
                    rows = []
            if rows:
                ROWS_OK.labels(partition=p).inc(self.bulk_insert(rows))

            DURATION.labels(partition=p).observe(time.time() - t0)
            if latest_ts != ckpt:
                ckpt = latest_ts
                self.save_checkpoint(p, ckpt)

            lag = (datetime.now(tz=timezone.utc)
                   - datetime.fromisoformat(ckpt)).total_seconds()
            LAG_SEC.labels(partition=p).set(lag)
            time.sleep(self.cfg.get('poll_interval', 30))

        self.zk.stop(); self.pg.close()
        logging.info("Shutdown complete")


if __name__ == '__main__':
    logging.basicConfig(level=logging.INFO,
        format='%(asctime)s [%(levelname)s] %(message)s')
    ap = argparse.ArgumentParser()
    ap.add_argument('--config',    required=True)
    ap.add_argument('--worker-id', type=int, required=True)
    args = ap.parse_args()
    with open(args.config) as f: cfg = yaml.safe_load(f)
    ETLWorker(cfg, args.worker_id).run()

#!/usr/bin/env python3
"""
inventory/dynamic_inventory.py
Reads hosts at runtime from a PostgreSQL CMDB.
Usage: ansible-playbook site.yml -i inventory/dynamic_inventory.py
"""
import json, os, sys
import psycopg2

def get_inventory():
    conn = psycopg2.connect(
        host=os.environ['CMDB_HOST'],
        dbname=os.environ['CMDB_DB'],
        user=os.environ['CMDB_USER'],
        password=os.environ['CMDB_PASS'],
    )
    cur = conn.cursor()
    cur.execute("""
        SELECT hostname, ip_address, role, environment, extra_vars
        FROM hosts
        WHERE status = 'active'
          AND environment = %s
        ORDER BY role, hostname
    """, (os.environ.get('ENV', 'production'),))

    groups, hostvars = {}, {}
    for hostname, ip, role, env, extra in cur.fetchall():
        groups.setdefault(role, {'hosts': []})['hosts'].append(hostname)
        groups.setdefault(env,  {'hosts': []})['hosts'].append(hostname)
        hv = {'ansible_host': ip}
        if extra: hv.update(extra)
        hostvars[hostname] = hv

    conn.close()
    return {
        '_meta': {'hostvars': hostvars},
        **groups,
        'all': {'children': list(groups.keys())},
    }

if __name__ == '__main__':
    if '--list' in sys.argv:
        print(json.dumps(get_inventory(), indent=2))
    elif '--host' in sys.argv:
        h = sys.argv[sys.argv.index('--host') + 1]
        print(json.dumps(
            get_inventory()['_meta']['hostvars'].get(h, {}), indent=2))
    else:
        sys.exit("Usage: --list | --host HOSTNAME")

// pipeline-manager.mjs — Node.js orchestration
import { execa } from 'execa';
import { Client } from '@elastic/elasticsearch';
import zkLib from 'node-zookeeper-client';
import pg from 'pg';
import { readFileSync } from 'fs';

const { Pool } = pg;

// ── Run Ansible playbook ──────────────────────────────────────
async function runPlaybook(playbook, opts = {}) {
  const args = [playbook];
  if (opts.tags?.length)       args.push('--tags', opts.tags.join(','));
  if (opts.extraVars)          args.push('-e', JSON.stringify(opts.extraVars));
  if (opts.limit)              args.push('--limit', opts.limit);
  if (process.env.VAULT_PW_FILE)
    args.push('--vault-password-file', process.env.VAULT_PW_FILE);

  const proc = execa('ansible-playbook', args, {
    cwd: new URL('../ansible', import.meta.url).pathname,
    env: { ...process.env, ANSIBLE_FORCE_COLOR: '1' },
    all: true,
  });
  proc.all.pipe(process.stdout);
  try   { await proc; return { ok: true }; }
  catch { return { ok: false, msg: proc.shortMessage }; }
}

// ── Elasticsearch health check ────────────────────────────────
async function esHealth() {
  const es = new Client({
    nodes: [process.env.ES_NODE],
    auth: { username: process.env.ES_USER, password: process.env.ES_PASS },
    tls: { ca: readFileSync(process.env.ES_CA_CERT) },
  });
  const [h, s] = await Promise.all([
    es.cluster.health({ wait_for_status: 'yellow', timeout: '10s' }),
    es.cluster.stats(),
  ]);
  return {
    status:   h.status,
    nodes:    h.number_of_nodes,
    shards:   h.active_shards,
    docsM:    (s.indices.docs.count / 1e6).toFixed(1) + 'M',
    storeGB:  (s.indices.store.size_in_bytes / 1e9).toFixed(1) + 'GB',
  };
}

// ── ZooKeeper partition status ────────────────────────────────
function zkStatus(connectStr) {
  return new Promise((resolve, reject) => {
    const zk = zkLib.createClient(connectStr);
    zk.once('connected', () => {
      zk.getChildren('/etl/workers', null, (err, children) => {
        zk.close();
        if (err) return reject(err);
        resolve({
          partitions: (children || []).filter(c => c.startsWith('partition-')).length,
          checkpoints: (children || []).filter(c => c.startsWith('ckpt-')).length,
        });
      });
    });
    zk.connect();
    setTimeout(() => { zk.close(); reject(new Error('ZK timeout')); }, 8000);
  });
}

// ── DB watermarks ─────────────────────────────────────────────
async function watermarks() {
  const pool = new Pool({
    host: process.env.PG_HOST, database: 'etl',
    user: process.env.PG_USER, password: process.env.PG_PASS,
  });
  const res = await pool.query(`
    SELECT partition_id,
           ROUND(EXTRACT(EPOCH FROM (now()-last_timestamp))/60,1) AS lag_min,
           records_processed
    FROM etl_watermarks ORDER BY partition_id`);
  await pool.end();
  return res.rows;
}

// ── Health dashboard ──────────────────────────────────────────
async function healthDashboard() {
  console.log('\n══ Pipeline Health ══\n');
  const [es, zk, wm] = await Promise.allSettled([
    esHealth(), zkStatus(process.env.ZK_CONNECT), watermarks(),
  ]);
  if (es.status === 'fulfilled') {
    const icon = { green:'🟢', yellow:'🟡', red:'🔴' }[es.value.status] ?? '⚪';
    console.log(`ES: ${icon} ${es.value.status.toUpperCase()}`);
    console.log(`    Nodes:${es.value.nodes} Shards:${es.value.shards} Docs:${es.value.docsM} Store:${es.value.storeGB}`);
  }
  if (zk.status === 'fulfilled') {
    console.log(`\nZK: Active partitions: ${zk.value.partitions}, Checkpoints: ${zk.value.checkpoints}`);
  }
  if (wm.status === 'fulfilled') {
    console.log('\nPartition Lag:');
    wm.value.forEach(r => {
      const icon = r.lag_min > 60 ? '🔴' : '🟢';
      console.log(`  P${r.partition_id}: ${icon} ${r.lag_min}min lag  (${r.records_processed} processed)`);
    });
  }
  console.log('\n' + '═'.repeat(24));
}

else if (cmd === 'health') await healthDashboard();
else console.log('Usage: node pipeline-manager.mjs deploy | health');
