ZooKeeper Setup

Apache ZooKeeper is a distributed coordination service used for leader election, distributed locks, configuration management, and service discovery. In an ETL pipeline, ZooKeeper coordinates partition ownership so workers don't process duplicate data.

⚡ Leader Election
Automatically elects a leader node in distributed systems
🔒 Distributed Locks
Prevents race conditions across ETL workers
⚙️ Config Store
Central, consistent configuration across nodes
🗺️ Service Discovery
Nodes register ephemeral znodes on startup

python 
# src/worker.py
import argparse, yaml, json, logging
from kazoo.client import KazooClient
from kazoo.exceptions import NodeExistsError
from elasticsearch import Elasticsearch
import psycopg2

def claim_partition(zk, worker_id, root):
    """Claim the lowest unclaimed partition via ZooKeeper ephemeral node."""
    for partition in range(100):
        path = f"{root}/partition-{partition:04d}"
        try:
            zk.create(path, value=str(worker_id).encode(),
                      ephemeral=True, makepath=True)
            logging.info(f"Worker {worker_id} claimed partition {partition}")
            return partition
        except NodeExistsError:
            continue
    raise RuntimeError("No free partitions available")

def scroll_elasticsearch(es, cfg, query):
    """Yield batches of documents using the Scroll API."""
    resp = es.search(
        index=cfg["index"],
        body=query,
        scroll=cfg["scroll_timeout"],
        size=cfg["scroll_size"],
    )
    scroll_id = resp["_scroll_id"]
    hits = resp["hits"]["hits"]
    while hits:
        yield hits
        resp = es.scroll(scroll_id=scroll_id,
                         scroll=cfg["scroll_timeout"])
        scroll_id = resp["_scroll_id"]
        hits = resp["hits"]["hits"]
    es.clear_scroll(scroll_id=scroll_id)

def transform(doc):
    src = doc["_source"]
    return {
        "ts":         src["@timestamp"],
        "event_type": src.get("event_type", "unknown"),
        "user_id":    src.get("user_id"),
        "payload":    json.dumps(src.get("payload", {})),
    }

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--worker-id", type=int, required=True)
    parser.add_argument("--config",    required=True)
    parser.add_argument("--query",     required=True)
    args = parser.parse_args()

    with open(args.config) as f: cfg = yaml.safe_load(f)
    with open(args.query)  as f: query = json.load(f)

    zk = KazooClient(hosts=cfg["zookeeper"]["hosts"])
    zk.start()
    partition = claim_partition(
        zk, args.worker_id, cfg["zookeeper"]["root_path"])

    # Inject partition shard filter into ES query
    query["query"]["bool"]["filter"].append(
        {"term": {"partition": partition}})

    es = Elasticsearch(
        cfg["elasticsearch"]["hosts"],
        http_auth=(cfg["elasticsearch"]["username"],
                   cfg["elasticsearch"]["password"]),
        use_ssl=cfg["elasticsearch"]["use_ssl"],
        ca_certs=cfg["elasticsearch"]["ca_cert"],
    )

    conn = psycopg2.connect(
        host=cfg["destination"]["host"],
        port=cfg["destination"]["port"],
        dbname=cfg["destination"]["database"],
        user=cfg["destination"]["username"],
        password=cfg["destination"]["password"],
    )

    with conn.cursor() as cur:
        for batch in scroll_elasticsearch(
                es, cfg["elasticsearch"], query):
            rows = [transform(doc) for doc in batch]
            cur.executemany(
                "INSERT INTO events "
                "(ts, event_type, user_id, payload) "
                "VALUES (%(ts)s,%(event_type)s,"
                "%(user_id)s,%(payload)s) "
                "ON CONFLICT DO NOTHING",
                rows)
            conn.commit()
            logging.info(
                f"Loaded {len(rows)} rows "
                f"(partition {partition})")
    zk.stop()

if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO,
        format="%(asctime)s [%(levelname)s] %(message)s")
    main()

Use Node.js to trigger Ansible, query Elasticsearch, and interact with ZooKeeper from your CI/CD pipeline or monitoring tools.

javascript 
// deploy.js — trigger Ansible from Node.js CI
const { execFile } = require('child_process');
const path = require('path');

/**
 * Run an Ansible playbook programmatically.
 * @param {string} playbook - path to .yml playbook
 * @param {object} options  - extra vars, tags, limit
 */
function runPlaybook(playbook, options = {}) {
  return new Promise((resolve, reject) => {
    const args = [playbook];

    if (options.extraVars) {
      args.push('-e', JSON.stringify(options.extraVars));
    }
    if (options.tags) {
      args.push('--tags', options.tags.join(','));
    }
    if (options.limit) {
      args.push('--limit', options.limit);
    }
    if (process.env.VAULT_PASSWORD_FILE) {
      args.push('--vault-password-file',
                process.env.VAULT_PASSWORD_FILE);
    }

    const proc = execFile('ansible-playbook', args, {
      cwd: path.resolve(__dirname, '../ansible'),
      env: { ...process.env, ANSIBLE_FORCE_COLOR: '1' },
    });

    proc.stdout.on('data', d => process.stdout.write(d));
    proc.stderr.on('data', d => process.stderr.write(d));

    proc.on('close', code => {
      code === 0
        ? resolve({ success: true, code })
        : reject(new Error(`Playbook exited with code ${code}`));
    });
  });
}

// Usage: deploy only ETL workers with increased batch size
runPlaybook('site.yml', {
  tags: ['etl'],
  extraVars: { etl_batch_size: 2000 },
})
  .then(() => console.log('✅ ETL deployment complete'))
  .catch(err => { console.error('❌', err.message); process.exit(1); });

// es-client.js — Elasticsearch scroll consumer
const { Client } = require('@elastic/elasticsearch');

const es = new Client({
  nodes: ['https://es1.example.com:9200',
          'https://es2.example.com:9200'],
  auth: {
    username: process.env.ES_USER,
    password: process.env.ES_PASS,
  },
  tls: { ca: require('fs').readFileSync('/etc/ssl/es-ca.crt') },
});

/**
 * Async generator: scroll through all matching docs.
 */
async function* scrollDocs(index, query, batchSize = 500) {
  const { body: init } = await es.search({
    index,
    scroll: '2m',
    size: batchSize,
    body: query,
  });

  let scrollId = init._scroll_id;
  let hits = init.hits.hits;

  while (hits.length > 0) {
    yield hits;
    const { body: next } = await es.scroll({
      scroll_id: scrollId,
      scroll: '2m',
    });
    scrollId = next._scroll_id;
    hits = next.hits.hits;
  }

  await es.clearScroll({ body: { scroll_id: scrollId } });
}

// Example: count events by type over last 24 hours
async function aggregateByType() {
  const query = {
    query: {
      range: {
        '@timestamp': { gte: 'now-1d', lt: 'now' },
      },
    },
  };

  const counts = {};
  for await (const batch of scrollDocs('events-*', query)) {
    for (const hit of batch) {
      const type = hit._source.event_type ?? 'unknown';
      counts[type] = (counts[type] ?? 0) + 1;
    }
  }
  console.table(counts);
}

aggregateByType().catch(console.error);

// zk-monitor.js — watch ETL worker partitions
const zookeeper = require('node-zookeeper-client');

const ZK_HOSTS = 'zk1:2181,zk2:2181,zk3:2181';
const ETL_ROOT = '/etl/workers';

const client = zookeeper.createClient(ZK_HOSTS, {
  sessionTimeout: 10000,
  retries: 3,
});

function watchPartitions() {
  client.getChildren(
    ETL_ROOT,
    // Watcher fires on any children change
    event => {
      console.log(`[ZK] ${event.type} on ${event.path}`);
      watchPartitions(); // re-register watcher
    },
    (err, children) => {
      if (err) {
        console.error('[ZK] Error:', err);
        return;
      }
      console.log(
        `[ZK] Active partitions (${children.length}):`,
        children.sort()
      );
    }
  );
}

client.once('connected', () => {
  console.log('[ZK] Connected to', ZK_HOSTS);
  watchPartitions();
});

client.connect();

// Graceful shutdown
process.on('SIGINT', () => {
  client.close();
  process.exit(0);
});
